6    import java.io.IOException;

10   @WebServlet(name = "OperationsServlet", value = "/OperationsServlet", urlPatterns = {"/Q19/OperationsServlet"})
11   public class OperationsServlet extends HttpServlet {
12       @Override
13       protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
14           getServletContext().getRequestDispatcher("/operate.jsp").forward(request,response);
15       }
16   
17       @Override
18       protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
19           String num1 = request.getParameter("num1");
20           String num2 = request.getParameter("num2");
21           String operator = request.getParameter("operator");
22   
23           int n1 = Integer.parseInt(num1);
24           int n2 = Integer.parseInt(num2);
25           double answer = 0;
26           String operation = null;
27           if (Objects.equals(operator, "+")) {
28               operation = "ADDITION";
29               answer = n1 + n2;
30           } else if (Objects.equals(operator, "-")) {
31               operation = "SUBTRACTION";
32               answer = n1 - n2;
33           } else if (Objects.equals(operator, "*")) {
34               operation = "MULTIPLICATION";
35               answer = n1*n2;
36           } else {
37               operation = "DIVISION";
38               answer = n1/n2;
39           }
40           request.setAttribute("operation", operation);
41           request.setAttribute("operator", operator);
42           request.setAttribute("answer", answer);
43           doGet(request, response);
44   
45       }
